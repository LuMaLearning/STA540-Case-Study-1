# ---- helpers ----
fmt_n_pct <- function(n, denom) sprintf("%d (%.1f)", n, 100 * n / denom)
fmt_med_iqr <- function(med, q1, q3) sprintf("%.0f (%.0f–%.0f)", med, q1, q3) # tweak decimals if needed

# ---- denominators ----
N_all <- nrow(df)

# ---- build Table 1 rows (Characteristic + Value + section + indent) ----
table1_df <- bind_rows(

  # Ethnicity, n (%)
  tibble(
    section = "Ethnicity, n (%)",
    Characteristic = "Hispanic/Latinx",
    Value = fmt_n_pct(sum(df$Q5_1 == 1, na.rm = TRUE), N_all),
    indent = 1
  ),

  # Race, n (%)
  df %>%
    mutate(
      Q5_3_chr = as.character(Q5_3),
      race_clean = case_when(
        str_detect(Q5_3_chr, ",") ~ "Multiracial",
        Q5_3_chr == "23" ~ "White",
        Q5_3_chr == "24" ~ "Black or African American",
        Q5_3_chr == "25" ~ "American Indian or Alaska Native",
        Q5_3_chr == "26" ~ "Asian",
        Q5_3_chr == "27" ~ "Native Hawaiian or Pacific Islander",
        Q5_3_chr == "28" ~ "Other",
        Q5_3_chr == ""   ~ "Other",
        TRUE ~ NA_character_
      )
    ) %>%
    filter(!is.na(race_clean)) %>%
    count(race_clean, name = "n") %>%
    mutate(
      section = "Race, n (%)",
      Characteristic = race_clean,
      Value = fmt_n_pct(n, N_all),
      indent = 1
    ) %>%
    select(section, Characteristic, Value, indent),

  # History of PrEP uptake, n (%)
  df %>%
    mutate(
      prep_clean = case_when(
        Q6_2 == 1 ~ "In the past 6 months",
        Q6_2 == 3 ~ "Never taken PrEP",
        TRUE ~ NA_character_
      )
    ) %>%
    filter(!is.na(prep_clean)) %>%
    count(prep_clean, name = "n") %>%
    mutate(
      section = "History of PrEP uptake, n (%)",
      Characteristic = prep_clean,
      Value = fmt_n_pct(n, N_all),
      indent = 1
    ) %>%
    select(section, Characteristic, Value, indent),

  # Condom use frequency, n (%)
  df %>%
    mutate(
      condom_clean = case_when(
        Q11_3 == 1 ~ "Never",
        Q11_3 == 2 ~ "Sometimes",
        Q11_3 == 3 ~ "About half the time",
        Q11_3 == 4 ~ "Most of the time",
        Q11_3 == 5 ~ "Always",
        TRUE ~ NA_character_
      )
    ) %>%
    filter(!is.na(condom_clean)) %>%
    count(condom_clean, name = "n") %>%
    mutate(
      section = "Condom use frequency, n (%)",
      Characteristic = condom_clean,
      Value = fmt_n_pct(n, N_all),
      indent = 1
    ) %>%
    select(section, Characteristic, Value, indent),

  # Male sexual partners in past 90 days, median (IQR)
  df %>%
    summarise(
      med = median(Q11_2, na.rm = TRUE),
      q1  = quantile(Q11_2, 0.25, na.rm = TRUE),
      q3  = quantile(Q11_2, 0.75, na.rm = TRUE)
    ) %>%
    transmute(
      section = NA_character_,
      Characteristic = "Male sexual partners in past 90 days, median (IQR)",
      Value = fmt_med_iqr(med, q1, q3),
      indent = 0
    ),

  # Condomless anal sex in past 90 days, n (%)
  tibble(
    section = NA_character_,
    Characteristic = "Condomless anal sex in past 90 days, n (%)",
    Value = fmt_n_pct(sum(df$Q11_4 == 1, na.rm = TRUE), N_all),
    indent = 0
  ),

  # HIV testing, n (%): lifetime tested vs never tested
  tibble(
    section = "HIV testing history, n (%)",
    Characteristic = c("Tested for HIV in lifetime", "Never tested for HIV"),
    Value = c(
      fmt_n_pct(sum(df$Q11_5 == 1, na.rm = TRUE), N_all),
      fmt_n_pct(sum(df$Q11_5 == 2, na.rm = TRUE), N_all)
    ),
    indent = 1
  ),

  # If never tested: MAIN reason not tested, n (%)
  df %>%
    mutate(
      not_tested_clean = case_when(
        Q11_7 == 1 ~ "Unlikely to be exposed to HIV",
        Q11_7 == 2 ~ "Afraid of testing HIV-positive",
        Q11_7 == 3 ~ "Did not want to think about HIV/HIV-positive",
        Q11_7 == 4 ~ "Worried about names being reported if positive",
        Q11_7 == 5 ~ "Dislike for needles",
        Q11_7 == 6 ~ "Unable to trust that the results will be confidential",
        Q11_7 == 7 ~ "Unaware of where to get tested",
        Q11_7 == 8 ~ "Other reasons",
        TRUE ~ NA_character_
      )
    ) %>%
    filter(!is.na(not_tested_clean)) %>%
    count(not_tested_clean, name = "n") %>%
    mutate(
      section = "If never tested for HIV: main reason, n (%)",
      Characteristic = not_tested_clean,
      Value = fmt_n_pct(n, N_all),
      indent = 1
    ) %>%
    select(section, Characteristic, Value, indent)
)
